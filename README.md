 #c) Check whether user name is palindrome or not.
 def palindrome(word):
     x=word.replace(" ","")
     return x==x[::-1]

 username = input("Enter your name:-")

 if palindrome(username):
     print(f"{username} is a palindrome!")
 else:
     print(f"{username} is not a palindrome.")
